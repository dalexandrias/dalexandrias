# 1️⃣ Obtém a versão atual - script: | VERSION_FILE="version.json" if [ -f "$VERSION_FILE" ]; then CURRENT_VERSION=$(jq -r '.version' "$VERSION_FILE") else CURRENT_VERSION="1.0.0" fi echo "Versão atual: $CURRENT_VERSION" echo "##vso[task.setvariable variable=CURRENT_VERSION]$CURRENT_VERSION" displayName: 'Obter versão atual do version.json' # 2️⃣ Gera a nova versão - script: | BUILD_NUMBER=$(echo $(CURRENT_VERSION) | awk -F. '{print $3}') NEW_BUILD_NUMBER=$((BUILD_NUMBER + 1)) NEW_VERSION=$(echo $(CURRENT_VERSION) | awk -F. '{print $1"."$2}')".$NEW_BUILD_NUMBER" echo "Nova versão: $NEW_VERSION" echo "##vso[task.setvariable variable=NEW_VERSION]$NEW_VERSION" displayName: 'Gerar nova versão' # 3️⃣ Atualiza o version.json - script: | TIMESTAMP=$(date +%Y%m%d-%H%M%S) echo "{ \"version\": \"$(NEW_VERSION)\", \"timestamp\": \"$TIMESTAMP\" }" > version.json displayName: 'Atualizar version.json com nova versão' # Copia o build para a pasta /out com a nova versão - script: | mkdir -p out/build-$(NEW_VERSION) cp -r dist/* out/build-$(NEW_VERSION) displayName: 'Salvar build na pasta /out' # Faz commit das alterações e envia para o repositório - script: | git config --global user.email "$(Build.RequestedFor)@yourdomain.com" git config --global user.name "$(Build.RequestedFor)" git add out/ version.json git commit -m "Gerando nova versão do build: $(NEW_VERSION) - Commit feito por $(Build.RequestedFor)" git push origin main displayName: 'Commit e push da nova versão' env: SYSTEM_ACCESSTOKEN: $(System.AccessToken)
